<script>
  const urlParams = new URLSearchParams(window.location.search);
  const name = urlParams.get('name'); // Wert des Parameters 'name'

  if (name) {
      // Den Namen auf der Seite anzeigen
      document.getElementById('info').innerText = `Willkommen, ${name}!`;
  } else {
      document.getElementById('info').innerText = "Kein Name angegeben!";
  
</script>
